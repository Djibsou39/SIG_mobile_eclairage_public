CREATE TABLE IF NOT EXISTS public.photo
(
    gid SERIAL PRIMARY KEY,
	photo character varying(40) COLLATE pg_catalog."default",
    idpl integer,
    CONSTRAINT fk_photo_idpl FOREIGN KEY (idpl)
        REFERENCES public.pl (gid) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE CASCADE,
    CONSTRAINT fk_photo_idpl_not_null CHECK (idpl IS NOT NULL)
)

TABLESPACE pg_default;
