DROP TABLE IF EXISTS public.pl;

CREATE TABLE IF NOT EXISTS public.pl
(
    gid serial PRIMARY KEY,  -- Utilisez le type serial pour générer automatiquement les valeurs
    geom geometry(Point, 2154),
    id bigint,
    codecoffret varchar(20),  -- Utilisez varchar plutôt que character varying
    codecable varchar(20),
    datereleve varchar(30),
    typepl integer,
    materiau integer,
    nblampes integer,
    etatpl integer,
    remarque varchar(109),
    nomrue varchar(50),
    photo varchar(40),
	x_wgs84 numeric,
	y_wgs84 numeric,
	x_l93 numeric,
	y_l93 numeric,
    lienstreetview varchar(200)
);
